<script>
// 日英の言語テキスト定義
const TRANSLATIONS = {

walkingMode: "歩行瞑想中...", sittingMode: "坐位瞑想中..."
},
en: {

walkingText: "Walking", sittingText: "Sitting", errorTime: "Please enter a valid time.",
startMsg: " meditation session started. Let's begin with walking meditation.",
switchMsg: "Walking meditation finished. Next is sitting meditation. Please take your seat.",
endMsg: "The session has ended. Sadhu, Sadhu, Sadhu.",

}
};

const STAGES = {
1: {
name: { ja: "第1段階", en: "Stage 1" },
walking: { ja: ["右足", "左足"], en: ["Right foot", "Left foot"] },
sitting: { ja: ["膨らみ", "縮み"], en: ["Rising", "Falling"] }
},
2: {
name: { ja: "第2段階", en: "Stage 2" },
walking: { ja: ["上げる", "置く"], en: ["Lifting", "Placing"] },
sitting: { ja: ["膨らみ", "縮み"], en: ["Rising", "Falling"] }
},
3: {
name: { ja: "第3段階", en: "Stage 3" },
walking: { ja: ["上げる", "運ぶ", "置く"], en: ["Lifting", "Moving", "Placing"] },
sitting: { ja: ["膨らみ", "縮み", "座っている", "触れている"], en: ["Rising", "Falling", "Sitting", "Touching"] }
},
4: {
name: { ja: "第4段階", en: "Stage 4" },
walking: { ja: ["踵を上げる", "上げる", "運ぶ", "置く"], en: ["Raising heel", "Lifting", "Moving", "Placing"] },
sitting: { ja: ["膨らみ", "縮み", "座っている", "触れている"], en: ["Rising", "Falling", "Sitting", "Touching"] }
},
5: {
name: { ja: "第5段階", en: "Stage 5" },
walking: { ja: ["踵を上げる", "上げる", "運ぶ", "下ろす", "置く"], en: ["Raising heel", "Lifting", "Moving", "Lowering", "Placing"] },
sitting: { ja: ["膨らみ", "縮み", "座っている", "触れている"], en: ["Rising", "Falling", "Sitting", "Touching"] }
},
6: {
name: { ja: "第6段階", en: "Stage 6" },

sitting: { ja: ["膨らみ", "縮み", "座っている", "触れている"], en: ["Rising", "Falling", "Sitting", "Touching"] }
}
};

let lang = 'ja';
let currentStage = 1;
let isRunning = false;
let mode = '待機';
let timeLeft = 0;
let timerInterval = null;
let speechInterval = null;

function setLanguage(l) {
if (isRunning) return;
lang = l;
document.getElementById('btn-ja').classList.toggle('active', lang === 'ja');
document.getElementById('btn-en').classList.toggle('active', lang === 'en');

// UIテキストの更新
const t = TRANSLATIONS[lang];
document.getElementById('label-stage').innerText = t.stage;
document.getElementById('label-time').innerText = t.time;
document.getElementById('btn-start').innerText = t.start;
document.getElementById('btn-stop').innerText = t.stop;
document.getElementById('label-current').innerText = t.current;

updateGuide();
}

function changeStage(stage) {
if (isRunning) return;
currentStage = stage;
document.querySelectorAll('.stage-btn').forEach((btn, idx) => {
btn.classList.toggle('active', idx === stage - 1);
});
updateGuide();
}

function updateGuide() {
const g = STAGES[currentStage];
const t = TRANSLATIONS[lang];
document.getElementById('labelGuide').innerHTML = 【${t.walkingText}】 ${g.walking[lang].join(' → ')}&lt;br&gt;【${t.sittingText}】 ${g.sitting[lang].join(' → ')};
}

function speak(text, callback) {
window.speechSynthesis.cancel();
const uttr = new SpeechSynthesisUtterance(text);
uttr.lang = lang === 'ja' ? 'ja-JP' : 'en-US';
uttr.rate = lang === 'ja' ? 0.8 : 0.75; // 英語は少しゆったりめ
if (callback) uttr.onend = callback;
window.speechSynthesis.speak(uttr);
}

function startSpeechLoop(labels) {
clearInterval(speechInterval);
let index = 0;
speak(labels[index]);
index = (index + 1) % labels.length;

speechInterval = setInterval(() => {
speak(labels[index]);
index = (index + 1) % labels.length;
}, 5000); // 5秒間隔
}

function startMeditation() {
const mins = parseInt(document.getElementById('durationInput').value);
if (isNaN(mins) || mins <= 0) return alert(TRANSLATIONS[lang].errorTime);

isRunning = true;
mode = '歩行';
timeLeft = mins * 60;

document.getElementById('setupArea').style.display = 'none';
document.getElementById('statusArea').style.display = 'block';
document.getElementById('modeText').innerText = TRANSLATIONS[lang].walkingMode;
updateTimerDisplay();

const welcomeMsg = lang === 'ja'
? ${STAGES[currentStage].name[lang]}${TRANSLATIONS[lang].startMsg}
: ${STAGES[currentStage].name[lang]} ${TRANSLATIONS[lang].startMsg};

speak(welcomeMsg, () => {
if(isRunning && mode === '歩行') startSpeechLoop(STAGES[currentStage].walking[lang]);
});

timerInterval = setInterval(() => {
timeLeft--;
updateTimerDisplay();
if (timeLeft <= 0) nextPhase();
}, 1000);
}

function nextPhase() {
clearInterval(speechInterval);
if (mode === '歩行') {
mode = '坐位';
const mins = parseInt(document.getElementById('durationInput').value);
timeLeft = mins * 60;
document.getElementById('modeText').innerText = TRANSLATIONS[lang].sittingMode;
updateTimerDisplay();

speak(TRANSLATIONS[lang].switchMsg);
setTimeout(() => {
if (isRunning && mode === '坐位') startSpeechLoop(STAGES[currentStage].sitting[lang]);
}, 12000); // 英語のアナウンス長さを考慮して12秒の猶予
} else {
stopMeditation();
speak(TRANSLATIONS[lang].endMsg);
}
}

function stopMeditation() {
isRunning = false;
clearInterval(timerInterval);
clearInterval(speechInterval);
window.speechSynthesis.cancel();
document.getElementById('setupArea').style.display = 'block';
document.getElementById('statusArea').style.display = 'none';
}

function updateTimerDisplay() {
const m = Math.floor(timeLeft / 60);
const s = timeLeft % 60;
document.getElementById('timerText').innerText = ${m}:${s.toString().padStart(2, '0')};
}

// 初期表示の更新
updateGuide();
